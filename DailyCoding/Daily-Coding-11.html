# 크레인

문제 주어진 맵에서 인형뽑기의 움직임(moves)에 따라 인형을 뽑고 바구니에 옮겨서 저장한다. 바구니에서 같은 인형이 만나서 터진 인형의 갯수를 반환하는 문제이다.

접근 인형을 저장한 자료구조 필요. 바구니의 구조를 봤을 때 스택이 적절해 보임 뽑기의 움직임에 따라 주어진 board를 탐색하고 스택에 넣고 빼고의 요구사항 처리

function solution(borad, moves) {
  let answer = 0;

  let pick = [];
  moves.forEach(index => {
    const doll = pickup(board, index - 1);
    if (doll) {
      pick.push(doll);
    }
  });
  answer = pick.length - check(pick).length;
  return answer;
}
function check(pick) {
  const disapear = pick.filter((a, b) => {
    if (a === pick[b + 1] || a === pick[b - 1]) {
      return false;
    }
    return true;
  });
  if (disapear.length === pick.length) return disapear;
  return check(disapear);
}
function pickup(board, index) {
  for (let i = 0; i < board.length; i++) {
    if (board[i][index] !== 0) {
      const doll = board[i][index];
      board[i][index] = 0;
      return doll;
    }
  }
}
const board = [
  [0, 0, 0, 0, 0],
  [0, 0, 1, 0, 3],
  [0, 2, 5, 0, 1],
  [4, 2, 4, 4, 2],
  [3, 5, 1, 3, 1]
];
const moves = [1, 5, 3, 5, 1, 2, 1, 4];
solution(board, moves);
